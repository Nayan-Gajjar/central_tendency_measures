# install.packages("psych")          # Install psych package
library("psych")

# generate 100 random numbers between 0 and 1000
randomlist = runif(100, min=0, max=1000)

# Arithmetic Mean  -------------------------------------------------------------
arithmetic_mean = mean(randomlist)
print(paste("Arithmetic Mean is.....", arithmetic_mean))

## [1] "Arithmetic Mean is..... 525.728291410487"

# Arithmetic mean using manual calculation
arithmetic_mean_manual = sum(randomlist) / length(randomlist)
print(paste("Manual Calculation of Arithmetic Mean is.....", arithmetic_mean_manual))

## [1] "Manual Calculation of Arithmetic Mean is..... 525.728291410487"

# Geometric Mean ---------------------------------------------------------------
geometric_mean_psych = psych::geometric.mean(randomlist)
print(paste("Geometric Mean is.....", geometric_mean_psych))

## [1] "Geometric Mean is..... 434.7343866604"

# Geometric mean using manual calculation
geometric_mean_manual = exp(mean(log(randomlist)))
print(paste("Manual Calculation of Geometric Mean is.....", geometric_mean_manual))

## [1] "Manual Calculation of Geometric Mean is..... 434.7343866604"

# Harmonic Mean ----------------------------------------------------------------
harmonic_mean_psych = psych::harmonic.mean(randomlist)
print(paste("Harmonic Mean is.....", harmonic_mean_psych))

## [1] "Harmonic Mean is..... 328.598360077798"

# Harmonic mean manual calculation ---------------------------------------------
harmonic_mean_manual = length(randomlist) / sum((randomlist)^(-1))
print(paste("Manual Calculation of Harmonic Mean is.....", harmonic_mean_manual))

## [1] "Manual Calculation of Harmonic Mean is..... 328.598360077798"

Mode 

# list
randomlist = list(1, 2, 3, 3, 4, 4, 4, 5, 5, 6)

# categorical data
cat_data = list("few", "few", "many", "some", "many")

# There is no inbuilt function
getmode <- function(v) {
  uniqv <- unique(v)
  uniqv[which.max(tabulate(match(v, uniqv)))]
}

mode_randomlist = getmode(randomlist)
print(paste("Mode of numbers is.....", mode_randomlist))

## [1] "Mode of numbers is..... 4"

# It will return "few" only instead of "few" and "many"
mode_cat_data = getmode(cat_data)
print(paste("Mode of categorical data is.....", mode_cat_data))

## [1] "Mode of categorical data is..... few"
